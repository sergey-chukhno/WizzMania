workerLoop() — Actor Pattern
void DatabaseManager::workerLoop() {
  while (true) {
    std::function<void()> task;
    {
      unique_lock<mutex> lock(m_mutex);
      m_cv.wait(lock, [this] {
        return m_stopWorker || !m_tasks.empty();
      });
      task = std::move(m_tasks.front());
      m_tasks.pop();
    }
    if (task) task();
  }
}

Anti SQL Injection

DatabaseManager.cpp

Toutes les requêtes utilisent des Prepared Statements avec sqlite3_bind_text(). Aucune concaténation de chaînes — les paramètres sont échappés automatiquement par SQLite.

createUser() — Prepared Statement
const char* sql =
  "INSERT INTO users"
  "(USERNAME, PASSWORD_HASH, SALT)"
  "VALUES (?, ?, ?)";  // ← Placeholders

sqlite3_stmt* stmt;
sqlite3_prepare_v2(m_db, sql, -1,
                   &stmt, nullptr);

// Bind user input safely
sqlite3_bind_text(stmt, 1,
  username.c_str(), -1, SQLITE_STATIC);
sqlite3_bind_text(stmt, 2,
  hash.c_str(), -1, SQLITE_STATIC);
sqlite3_bind_text(stmt, 3,
  salt.c_str(), -1, SQLITE_STATIC);

Hachage + Sel

DatabaseManager.cpp

Mots de passe hachés via SHA-256 (OpenSSL EVP API) combinés à un sel aléatoire 128-bit généré par RAND_bytes(). Jamais stockés en clair.

hashPassword() + generateSalt()
// Sel unique de 128 bits par utilisateur
std::string DatabaseManager::generateSalt() {
  unsigned char salt[16];
  RAND_bytes(salt, 16); // OpenSSL CSPRNG
  return bytesToHex(salt, 16);
}

std::string hashPassword(
    const string& pw, const string& salt) {
  string combined = pw + salt;
  EVP_MD_CTX* ctx = EVP_MD_CTX_new();
  EVP_DigestInit_ex(ctx, EVP_sha256(), nullptr);
  EVP_DigestUpdate(ctx, combined.c_str(),
                   combined.size());
  EVP_DigestFinal_ex(ctx, hash, &len);
  return bytesToHex(hash, len);
}

Transport TLS 1.2

TcpServer.cpp

Toutes les communications client–serveur sont encryptées via TLS 1.2. Le serveur charge un certificat X.509 et une clé privée. SSLv2 est désactivé explicitement.

TcpServer() — Configuration SSL
TcpServer::TcpServer(int port)
  : m_sslContext(asio::ssl::context::tlsv12) {

  m_sslContext.set_options(
    ssl::context::default_workarounds |
    ssl::context::no_sslv2 |  // ← Kill SSLv2
    ssl::context::single_dh_use);

  m_sslContext.use_certificate_chain_file(
    "server/certs/server.crt");

  m_sslContext.use_private_key_file(
    "server/certs/server.key",
    ssl::context::pem);
}

NativeSharedMemory

common/NativeSharedMemory.h

Wrapper cross-platform pour POSIX shm_open / mmap. Le jeu crée le segment, le client Qt l'ouvre en lecture. Sémaphore POSIX pour la synchronisation.

createAndMap() — POSIX Writer
bool NativeSharedMemory::createAndMap() {
  string shmName = "/" + m_name;
  m_fd = shm_open(shmName.c_str(),
    O_CREAT | O_RDWR, 0666);

  ftruncate(m_fd, sizeof(GameIPCData));

  m_data = (GameIPCData*) mmap(
    NULL, sizeof(GameIPCData),
    PROT_READ | PROT_WRITE,
    MAP_SHARED, m_fd, 0);

  m_sem = sem_open(
    semName.c_str(), O_CREAT, 0666, 1);
  return true;
}

Broadcast Live Score

TcpServer.cpp

Le client poll la mémoire partagée et envoie un paquet GameStatus au serveur. Celui-ci le broadcast à tous les followers en ligne du joueur.

handleGameStatus() — Broadcast
void TcpServer::handleGameStatus(
    ClientSession* sender,
    const string& gameName, uint32_t score) {

  Packet pkt(PacketType::GameStatus);
  pkt.writeString(sender->getUsername());
  pkt.writeString(gameName);
  pkt.writeInt(score);

  // Broadcast to all followers
  auto followers = m_db.getFollowers(username);
  for (const auto& f : followers) {
    auto it = m_onlineUsers.find(f);
    if (it != m_onlineUsers.end())
      it->second->sendPacket(pkt);
  }
}
